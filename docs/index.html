Quick Start

# Install
pip install bazinga-indeed

# Ask anything (no API keys needed!)
bazinga "What is consciousness?"

# Multi-AI consensus
bazinga --multi-ai "Is AI conscious?"

# Interactive chat
bazinga --chat

# Index your files for search
bazinga --index ~/Documents

# Search your knowledge base
bazinga --kb "find my research on phi"

# System check
bazinga --check
